如何使用 #

组件自带有常用的icon，使用简单高效。默认不内置，如需使用，请阅读开发指南，自定义引入iconfont样式
<template>
<hl-icon class="hl-icon_shuaxin"></hl-icon>
<hl-icon class="hl-icon_shuaxin" color="red" style="margin-left:15px;"></hl-icon>
<hl-icon class="hl-icon_shuaxin" size="26" style="margin-left:15px;"></hl-icon>
</template>
